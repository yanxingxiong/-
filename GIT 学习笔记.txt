git init //初始化本地库
git status // 查看本地库状态
git add 文件名 // 添加到暂存区
git commit -m "日志信息" 文件名  //提交到本地库
git reflog //查看历史记录
git reset --hard 版本号 // 版本穿梭
git rm --cached 文件名// 删除暂存区文件
git reflog // 查看版本日志信息
git log // 查看版本详细日志信息
git branch //查看分支
git branch 分支名字 //创建分支
git checkout 分支名字  // 切换分支
git merge 分支名字 // 把指定的分支合并到当前分支上
合并和提交解决冲突需要人为修改
git push 推送代码通关中心
